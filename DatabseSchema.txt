CREATE TABLE UserAccount(
	Title VARCHAR(4),
	Gender VARCHAR(20) NOT NULL,
	First_Name VARCHAR(20) NOT NULL,
	Last_Name VARCHAR(20) NOT NULL,
	Birthdate DATE NOT NULL,
	Company VARCHAR(40),
	Username VARCHAR(15) PRIMARY KEY,
	Password VARCHAR(15) NOT NULL,
	ProfileTagOne VARCHAR(20),
	ProfileTagTwo VARCHAR(20),
	ProfileTagThree VARCHAR(20),
	ProfileTagFour VARCHAR(20),
	ProfileTagFive VARCHAR(20),
	ProfileTagSix VARCHAR(20),
	Education VARCHAR(100),
	Location VARCHAR(30),
	Work_History VARCHAR(3000),
	Rating DECIMAL(2,1),
	ResumeURL VARCHAR(2048),
	WorkSampleURL VARCHAR(2048),
	Privacy BOOL,
	EmployerCommentsURL VARCHAR(2048)
	ProfileImageURL VARCHAR(2048));
	

CREATE TABLE Listings(
	ListingID INTEGER(15) PRIMARY KEY,
	HunterID VARCHAR(15),
	HunteeID VARCHAR(15),
	ListingTitle VARCHAR(20) NOT NULL,
	TagOne VARCHAR(10),
	TagTwo VARCHAR(10),
	TagThree VARCHAR(10),
	Description VARCHAR(2000) NOT NULL);

CREATE TABLE Payment(
	Username VARCHAR(15) NOT NULL,
	CardType VARCHAR(10) NOT NULL,
	CardNumber INTEGER(16) NOT NULL,
	NameOnCard VARCHAR(21) NOT NULL,
	ExpirationMonth INTEGER(2) NOT NULL,
	ExpirationYear INTEGER(2) NOT NULL,
	CVV INTEGER(4) NOT NULL,
	PRIMARY KEY (CardType, CardNumber));

CREATE TABLE Inbox(
	SenderUsername VARCHAR(15) NOT NULL,
	Message VARCHAR(2000),
	RecivUsername VARCHAR(15) NOT NULL,
	MessageID INTEGER(15) PRIMARY KEY));

CREATE TABLE ProgressTracker(
	ListingID INTEGER(12) PRIMARY KEY,
	CurrentProgress INTEGER(3) NOT NULL,
	UpdateDate DATE NOT NULL,
	MilestoneName VARCHAR(30));
